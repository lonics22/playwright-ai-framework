01

config/ - 配置管理

支持多环境配置（dev/staging/prod）、浏览器参数、AI 模型设置等 YAML 配置

02

core/ - 框架核心

驱动层封装、AI Agent 工厂、智能定位器、自定义工具等核心能力

03

pages/ - 页面对象

POM 设计模式，BasePage 提供传统方法，AIPage 扩展 AI 能力（ai_click, ai_fill）

04

tests/ - 测试用例

三种测试模式：traditional（传统）、ai_driven（纯 AI）、hybrid（混合，推荐）

05

reports/ - 测试报告

Allure 报告输出，包含 AI 执行日志、截图、视频、失败分析等丰富附件

import allure

@pytest.mark.smoke
@allure.feature("用户认证")

class TestLogin:

    @allure.title("使用有效凭据登录")
    def test_login_with_valid_credentials(self, page):

        login_page = LoginPage(page)
        login_page.navigate_to()

        # Act
        login_page.fill_username("user@example.com")
        login_page.fill_password("password123")
        login_page.click_login_button()

        # Assert
        assert login_page.is_logged_in()
        allure.attach(
            page.screenshot(),
            name="登录成功截图",
            attachment_type=allure.attachment_type.PNG
        )
tests/ai_driven/test_natural_language.py AI-Driven 
import pytest
import allure

@pytest.mark.ai
@allure.feature("AI 驱动测试")
@allure.story("自然语言任务执行")
class TestNaturalLanguage:
    """AI 模式：使用自然语言描述复杂场景"""

    @allure.title("完成完整购物流程")
    async def test_shopping_journey(self, ai_agent, page):
        # 传统方式：打开页面
        await page.goto("https://example-shop.com")

        # AI 方式：执行复杂任务
        result = await ai_agent.execute_task("""
            完成一次完整的购物流程：
            1. 在搜索框输入 "wireless headphones"
            2. 选择第一个商品加入购物车
            3. 进入购物车结算
            4. 使用测试信用卡完成支付
            5. 验证订单确认页面显示成功消息
        """)

        # Allure 附加 AI 执行详情
        allure.attach(
            result.execution_log,
            name="AI 执行日志",
            attachment_type=allure.attachment_type.TEXT
        )

        assert result.success

import pytest
import allure
from pages.login_page import LoginPage

@pytest.mark.hybrid
@allure.feature("混合模式测试")
class TestE2EWithAI:
    """混合模式：稳定流程用传统方式，易变流程用 AI"""

    @allure.title("登录后 AI 完成复杂操作")
    async def test_login_then_ai_task(self, page, ai_adapter):
        # 传统方式：稳定的登录流程（选择器固定、不易变）
        login_page = LoginPage(page)
        login_page.login("user@test.com", "password123")

        # AI 方式：处理复杂、易变的业务流程
        result = await ai_adapter.execute_task("""
            在个人中心完成以下操作：
            1. 找到订单历史标签并点击
            2. 找到最近一笔未评价的订单
            3. 为该订单添加五星好评，评论写 "Great product!"
            4. 上传一张测试图片作为评价配图
            5. 提交评价并确认成功
        """)

        assert result.success
pages/ai_page.py - 双模式页面对象基类 
from playwright.sync_api import Page
from core.driver.browser_use_adapter import AIAgentAdapter

class AIPage:
    """AI 增强的基础页面对象 - 支持传统和 AI 两种模式"""

    def __init__(self, page: Page, ai_adapter: AIAgentAdapter = None):
        self.page = page
        self.ai = ai_adapter
        self._smart_mode = ai_adapter is not None

    # ========== 传统模式方法 ==========
    def click(self, selector: str):
        """传统点击操作"""
        self.page.click(selector)

    def fill(self, selector: str, value: str):
        """传统输入操作"""
        self.page.fill(selector, value)

    # ========== AI 模式方法 ==========
    async def ai_click(self, element_description: str):
        """AI 驱动的点击 - 通过自然语言描述定位元素"""
        if not self._smart_mode:
            raise RuntimeError("AI adapter not initialized")

        task = f"在页面上找到 '{element_description}' 并点击它"
        return await self.ai.execute_task(task)

    async def ai_fill_form(self, form_data: dict):
        """AI 驱动的表单填写"""
        fields = ", ".join([f"{k}={v}" for k, v in form_data.items()])
        task = f"填写表单，字段值: {fields}"
        return await self.ai.execute_task(task)

    async def ai_verify(self, expectation: str):
        """AI 驱动的验证 - 自然语言断言"""
        task = f"验证页面是否符合预期: {expectation}"
        result = await self.ai.execute_task(task)
        return result.success

运行命令

常用测试执行命令速查

运行传统测试

pytest tests/traditional/ -v \
  --alluredir=reports/allure-results

执行所有传统脚本测试用例

运行 AI 驱动测试

pytest tests/ai_driven/ -v -m ai \
  --llm=openai \
  --alluredir=reports/allure-results

使用 OpenAI GPT-4 执行自然语言测试

运行混合测试

pytest tests/hybrid/ -v -m hybrid \
  --llm=anthropic \
  --alluredir=reports/allure-results

使用 Claude 模型执行混合模式测试

并发执行（传统测试）

pytest tests/traditional/ -n auto \
  --dist=loadgroup

使用 pytest-xdist 多进程并行执行

生成 Allure 报告

# 生成报告
allure generate reports/allure-results \
  -o reports/allure-report \
  --clean

# 启动报告服务
allure open reports/allure-report

生成并查看可视化测试报告
